    let current = 0;
    let score = 0;
    let answered = false;
    let userAnswers = [];

    const letters = ["A", "B", "C", "D"];

    const quizEl = document.getElementById("quiz");
    const resultEl = document.getElementById("result");
    const counterEl = document.getElementById("counter");
    const scoreLiveEl = document.getElementById("scoreLive");
    const questionEl = document.getElementById("question");
    const optionsEl = document.getElementById("options");
    const feedbackEl = document.getElementById("feedback");
    const nextBtn = document.getElementById("nextBtn");
    const restartBtn = document.getElementById("restartBtn");
    const restartBtn2 = document.getElementById("restartBtn2");
    const progressEl = document.getElementById("progress");
    const reviewEl = document.getElementById("review");
    const toggleReviewBtn = document.getElementById("toggleReviewBtn");

    function renderQuestion() {
      window.scrollTo({ top: 0, behavior: "smooth" });
      answered = false;
      nextBtn.disabled = true;
      feedbackEl.className = "feedback";
      feedbackEl.textContent = "";

      const q = questions[current];
      counterEl.textContent = `Pregunta ${current + 1} de ${questions.length}`;
      scoreLiveEl.textContent = `Aciertos: ${score}`;
      questionEl.textContent = q.q;
      optionsEl.innerHTML = "";

      const progress = (current / questions.length) * 100;
      progressEl.style.width = `${progress}%`;

      q.options.forEach((opt, index) => {
        const btn = document.createElement("button");
        btn.type = "button";
        btn.className = "option";
        btn.innerHTML = `<span class="letter">${letters[index]}</span><span>${opt}</span>`;
        btn.addEventListener("click", () => selectAnswer(index, btn));
        optionsEl.appendChild(btn);
      });
    }

    function selectAnswer(index, selectedBtn) {
      if (answered) return;
      answered = true;

      const q = questions[current];
      const isCorrect = index === q.answer;

      userAnswers[current] = index;

      if (isCorrect) {
        score++;
        selectedBtn.classList.add("correct");
        feedbackEl.className = "feedback ok";
        feedbackEl.textContent = "Correcto.";
      } else {
        selectedBtn.classList.add("incorrect");
        feedbackEl.className = "feedback bad";
        feedbackEl.textContent = `Incorrecto. Respuesta correcta: ${letters[q.answer]}) ${q.options[q.answer]}`;
      }

      [...optionsEl.children].forEach((child, i) => {
        child.classList.add("disabled");
        if (i === q.answer) child.classList.add("correct");
      });

      scoreLiveEl.textContent = `Aciertos: ${score}`;
      nextBtn.disabled = false;

      if (current === questions.length - 1) {
        nextBtn.textContent = "Ver resultado";
      }
    }

    function nextQuestion() {
      if (!answered) return;

      if (current < questions.length - 1) {
        current++;
        renderQuestion();
      } else {
        showResult();
      }
    }

    function showResult() {
      quizEl.style.display = "none";
      resultEl.style.display = "block";
      progressEl.style.width = "100%";

      const wrong = questions.length - score;
      const pct = Math.round((score / questions.length) * 100);

      document.getElementById("finalScore").textContent = `${score}/${questions.length}`;
      document.getElementById("totalRight").textContent = score;
      document.getElementById("totalWrong").textContent = wrong;
      document.getElementById("percent").textContent = `${pct}%`;

      let msg = "";
      if (pct >= 90) msg = "Excelente dominio de la LGSM.";
      else if (pct >= 80) msg = "Muy buen resultado. Revisa los errores puntuales.";
      else if (pct >= 70) msg = "Buen avance, pero todavía conviene reforzar figuras societarias.";
      else if (pct >= 60) msg = "Resultado regular. Repasa especialmente sociedades y figuras jurídicas.";
      else msg = "Necesitas reforzar los temas clave antes del examen.";

      document.getElementById("finalMessage").textContent = msg;
      buildReview();
    }

    function buildReview() {
      reviewEl.innerHTML = "";
      questions.forEach((q, i) => {
        const user = userAnswers[i];
        const ok = user === q.answer;
        const item = document.createElement("div");
        item.className = `review-item ${ok ? "right" : "wrong"}`;
        item.innerHTML = `
          <strong>${i + 1}. ${q.q}</strong>
          <div class="small">Tu respuesta: ${user !== undefined ? letters[user] + ") " + q.options[user] : "Sin responder"}</div>
          <div class="small">Correcta: ${letters[q.answer]}) ${q.options[q.answer]}</div>
        `;
        reviewEl.appendChild(item);
      });
    }

    function restart() {
      current = 0;
      score = 0;
      answered = false;
      userAnswers = [];
      nextBtn.textContent = "Siguiente";
      resultEl.style.display = "none";
      reviewEl.style.display = "none";
      toggleReviewBtn.textContent = "Ver revisión";
      quizEl.style.display = "block";
      renderQuestion();
    }

    [nextBtn, restartBtn, restartBtn2, toggleReviewBtn].forEach(btn => {
      if (btn) btn.type = "button";
    });

    nextBtn.addEventListener("click", nextQuestion);
    restartBtn.addEventListener("click", restart);
    restartBtn2.addEventListener("click", restart);
    toggleReviewBtn.addEventListener("click", () => {
      const visible = reviewEl.style.display === "block";
      reviewEl.style.display = visible ? "none" : "block";
      toggleReviewBtn.textContent = visible ? "Ver revisión" : "Ocultar revisión";
    });

    renderQuestion();
  </script>
